164int get_local_net_IP(char * client_IP)
165{
166 int result = 0;
167 // Retrieve the client's IP address
168 struct ifaddrs *ifap, *ifa;
169
170 if (getifaddrs(&ifap) == -1) {
171 perror("getifaddrs");
172 return FALSE;
173 }
174
175 size_t numLoopbackInterfaces = sizeof(loopbackInterfaceNames) / sizeof(loopbackInterfaceNames[0]);
176 for (ifa = ifap; ifa != NULL; ifa = ifa->ifa_next) {
177 if (ifa->ifa_addr == NULL || ifa->ifa_addr->sa_family != AF_INET) {
178 continue;
179 }
180
181 int isLoopback = 0;
182 for (size_t i = 0; i < numLoopbackInterfaces; ++i) {
183 if (strcmp(ifa->ifa_name, loopbackInterfaceNames[i]) == 0) {
184 isLoopback = 1;
185 break;
186 }
187 }
188
189 if (isLoopback) {
190 continue; // Skip loopback interface
191 }
192 struct sockaddr_in* addr = (struct sockaddr_in*)ifa->ifa_addr;
193 inet_ntop(AF_INET, &(addr->sin_addr), client_IP, INET_ADDRSTRLEN);
194 result = 1;
195 break;
196 }
197 freeifaddrs(ifap);
198 return result;
199}

568int isPortConvertableToUINT16(const char *str) {

570 if (strlen(str) > snprintf(NULL, 0, "%d", UINT16_MAX)) {
571 //The string has more characters than a string version of INT_MAX could have.
572 printf("This has too many characters to be a valid port number: %s",str);
573 return 0;
574 }
575 char *endptr;
576 long result = strtol(str, &endptr, 10);
577 // If endptr is equal to the input string, no conversion was performed
578 if (endptr == str)
579 {
580 printf("this is not a valid port number: %s",str);
581 return 0;
582 }
583 // If endptr points to a non-null character, there are trailing characters
584 if (*endptr != '\0' && *endptr != '\n')
585 {
586 printf("this is not a valid port number: %s",str);
587 return 0;
588 }
589 //check to see if the port can exist
590 if (result > UINT16_MAX || result < 0) {
591 printf("The value exceeds the maximum allowed port number.\n");
592 return 0;
