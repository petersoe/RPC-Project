20char* strtok_custom(char* str, const char* delimiters) {
21 static char* token = NULL;
22 if (str != NULL)
23 token = str;
24
25 if (token == NULL)
26 return NULL;
27
28 char* start = token;
29 char* end = strpbrk(token, delimiters);
30
31 if (end != NULL) {
32 *end = '\0';

35 token = NULL;

75int hash_string(char *str) {
76 int hash = 5381;
77 int c;
78 while (*str != '\0') {
79 c = *str++;
80 hash = ((hash << 5) + hash) + c; // djb2 hash function
81 }
82 return hash;

185int parse_Article(char *article, pubSub *sub, FILE *log) {
186 //strtok alters article, so we will create a duplicate.
187 char * articleNew = malloc(121 * sizeof (char));
188 strcpy(articleNew, article);
189 char* tokens[4];
190 char* token = strtok_custom(articleNew, ";");
191 int count = 0;
192 while (token != NULL) {
193 tokens[count] = token;
194 count++;
195 token = strtok_custom(NULL, ";");
196 }
197 if (count == 4) {
198 //convert to lower case (contents remains unchanged)
199 for (int i = 0; i < 3; i++) {
200 for (int j = 0; tokens[i][j]; j++) {
201 tokens[i][j] = tolower(tokens[i][j]);
202 }
203 }
204 bool_t isValidType = FALSE;
205 for (int j = 0; j < sizeof (valid_types) / sizeof (valid_types[0]); j++) {
206 if (strcmp(tokens[0], valid_types[j]) == 0) {
207 sub->type = strdup(tokens[0]);
208 isValidType = TRUE;
209 }
210 }
211 if (isValidType == FALSE) {
212 fprintf(log, "Error: first argument must be empty or one of the following types: Sports, Lifestyle, Entertainment, Business, Technology, Science, Politics, Health\n");
213 sub->isInvalid = TRUE;
214 return 0;
215 }
216 sub->originator = strdup(tokens[1]);
217 sub->org = strdup(tokens[2]);
218 sub->contents = strdup(tokens[3]);
219 } else if (count < 4) {
220 fprintf(log, "Error: expected 4 arguments in Article, received %d\n", count);
221 sub->isInvalid = TRUE;
222 return 0;
223 } else {
224 fprintf(log, "Error: expected 4 arguments in Article, received more than 4.\n", count);
225 sub->isInvalid = TRUE;
226 return 0;
227 }
228 if (strlen(sub->type) == 0 && strlen(sub->org) == 0 && strlen(sub->originator) == 0) {
229 fprintf(log, "Error: at least one of the first three arguments must not be empty.\n");
230 sub->isInvalid = TRUE;
231 return 0;
232 }
233 return 1;

869int saveFailedData(char * failedFilePath, pubSub ** FailedData, int failed_count, FILE *log) {
870 int failedFileDescriptor;
871 struct flock fl;
872 int lockAttempts = 0;
873 const int maxLockAttempts = 10;
874 const int lockWaitTimeSeconds = 0.5;
875 // Wait until file is unlocked or max attempts reached
876 while (lockAttempts < maxLockAttempts) {
877 failedFileDescriptor = open(failedFilePath, O_WRONLY | O_CREAT | O_TRUNC, 0644);
878 if (failedFileDescriptor >= 0) {
879 // Acquire exclusive lock
880 fl.l_type = F_WRLCK;
881 fl.l_whence = SEEK_SET;
882 fl.l_start = 0;
883 fl.l_len = 0;
884 if (fcntl(failedFileDescriptor, F_SETLK, &fl) == -1) {
885 fprintf(log, "Failed to acquire exclusive lock on file\n");
886 close(failedFileDescriptor);
887 return 0;
888 }
889
890 char numToSave[15];
891 sprintf(numToSave, "%d", failed_count);
892 dprintf(failedFileDescriptor, "%s\n", numToSave);
893 for (int i = 0; i < failed_count; i++) {
894 char portStr[6]; // assuming max 5 digits for the port
895 sprintf(portStr, "%d", (*FailedData)[i].client_address.port);
896 dprintf(failedFileDescriptor, "IP: %s | Port: %s | Contents: %s\n",
897 (*FailedData)[i].client_address.IP,
898 portStr,
899 (*FailedData)[i].contents);
900 }
901 // Release exclusive lock
902 fl.l_type = F_UNLCK;
903 if (fcntl(failedFileDescriptor, F_SETLK, &fl) == -1) {
904 fprintf(log, "Failed to release exclusive lock on file\n");
905 close(failedFileDescriptor);
906 return 0;
907 }
908 // Closing the file
909 close(failedFileDescriptor);
910 return 1;
911 }
912 // Wait for file to be unlocked
913 sleep(lockWaitTimeSeconds);
914 lockAttempts++;
915 }
916 fprintf(log, "Timed out waiting for exclusive lock on %s\n",failedFilePath);
917 return 0;
918}
